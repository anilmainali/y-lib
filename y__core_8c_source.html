142 long y_rand()
143 {
144  // Have we initialized the random seed yet?
145  static int _y_random_seed_initialized = 0;
146 
147  if(!_y_random_seed_initialized)
148  {
149  int seed, tm, rnd;
150  y_setup();
151 
152  // Seed the random number generator for later use.
153  // To make it random enough for our purposes mix in the vuser id and the adress of the vuser group name.
154  // In case the script itself already initialized the random number generator, use a random number from
155  // there as well.
156 
157  tm = (int)time(NULL);
158  rnd = rand();
159 
160  //lr_log_message("Seed values - time: %d, y_virtual_user_id: %d, y_virtual_user_group: %d, rand: %d", tm, y_virtual_user_id, (int)y_virtual_user_group, rnd);
161  seed = tm%10000 + y_virtual_user_id + ((int)y_virtual_user_group)%1000 + rnd%1000;
162  //lr_log_message("Initialising random seed: %d", seed);
163  srand( seed );
164  _y_random_seed_initialized = 1;
165  }
166 
167  {
168  // Because rand() does not return numbers above 32767 and we want to get at least 30 of the 31 bits
169  // of randomness that a long affords us we are going to roll multiple numbers and basically
170  // concatenate them together using bit shifts.
171  //
172  // ( If we were to go to 32 bits this function would return negative numbers, which would be undesirable
173  // because it will break people's expectations of what rand() does.)
174 
175  long result = rand() << 15 | rand();
176  //lr_log_message("y_rand: 30 random bits = %d, Y_RAND_MAX = %d", result, Y_RAND_MAX);
177 
178  // Doing a third call to rand() just to get 1 bit of entropy isn't really efficiënt ..
179  //result = (result << 1) | (rand() & 0x0000000000000001); // add another bit and we're done.
180  //lr_log_message("y_rand: final random roll = %x, Y_RAND_MAX = %d", result, Y_RAND_MAX);
181 
182  return result;
183  }
184 }

472 {
473  char* result;

475  size_t param_eval_size = strlen(param_name) +3; // parameter name + "{}" + '\0' (end of string)

477  //lr_log_message("y_cleanse_parameter(%s)", param_name );
478 
479  // Get the contents of the parameter using lr_eval_string_ext() - we can't use the
480  // regular version if we expect to find NULL in there.
481  snprintf( param_eval_string, param_eval_size, "{%s}", param_name );
482  lr_eval_string_ext(param_eval_string, param_eval_size-1, &result, &result_size, 0, 0, -1);
483  if( strcmp(param_eval_string, result) == 0 )
484  {
485  lr_error_message("y_get_cleansed_parameter: Parameter %s does not exist.", param_name);
486  lr_abort();

490  //lr_log_message("Cleansing param %s, result starts with '%-*.*s' and contains %d bytes.", param_name, result_size, result_size, result, result_size);
491  {
492  size_t result_strlen;
493  // Now replace NULL bytes (NULL) in the input with something else..
494  for( result_strlen = strlen(result); result_strlen < result_size; result_strlen = strlen(result))
495  {
