319 char* y_get_cleansed_parameter(const char* param_name, char replacement)
320 {
321  char* result;
322  unsigned long result_size;
323  size_t param_eval_size = strlen(param_name) +3; // parameter name + "{}" + '\0' (end of string)
324  char* param_eval_string = y_mem_alloc(param_eval_size);
325  //lr_log_message("y_cleanse_parameter(%s)", param_name );
326 
327  // Get the contents of the parameter using lr_eval_string_ext() - we can't use the
328  // regular version if we expect to find NULL in there.
329  snprintf( param_eval_string, param_eval_size, "{%s}", param_name );
330  lr_eval_string_ext(param_eval_string, param_eval_size-1, &result, &result_size, 0, 0, -1);
331  if( strcmp(param_eval_string, result) == 0 )
332  {
333  lr_error_message("y_get_cleansed_parameter: Parameter %s does not exist.", param_name);
334  lr_abort();
335  }
336  free(param_eval_string);
337 
338  //lr_log_message("Cleansing param %s, result starts with '%-*.*s' and contains %d bytes.", param_name, result_size, result_size, result, result_size);
339  {
340  size_t result_strlen;
341  // Now replace NULL bytes (NULL) in the input with something else..
342  for( result_strlen = strlen(result); result_strlen < result_size; result_strlen = strlen(result))
343  {
344  result[result_strlen] = replacement;
345  //lr_log_message("Cleansing param %s, result now '%-*.*s' and contains %d bytes.", param_name, result_size, result_size, result, result_size);
346  }
347  }
348  return result;
349 }
