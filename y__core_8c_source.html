151 long y_rand(void)
152 {
153  // Because rand() does not return numbers above 32767 and we want to get at least 30 of the 31 bits
154  // of randomness that a long affords us we are going to roll multiple numbers and basically
155  // concatenate them together using bit shifts.
156  //
157  // ( If we were to go to 32 bits this function would return negative numbers, which would be undesirable
158  // because it will break people's expectations of what rand() does.)
159  return rand() << 15 | rand();
160 }
