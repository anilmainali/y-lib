355 void y_save_transaction_end_status(char* transaction_name, const char* saveparam, int status)
356 {
357  int actual_status = lr_get_transaction_status(transaction_name);
358 
359  if( actual_status == LR_PASS )
360  {
361  // LR thinks everything is fine. If our code doesn't that becomes the new end status.
362  lr_set_transaction_status(status);
363  }
364  else // in case of a LR reported fail status of some kind.
365  {
366  // The loadrunner reported status takes precendence as those errors are usually quite fundamental.
367  status = actual_status;
368  }
369 
370  if( actual_status == -16863 )
371  {
372  // Fix me: Lookup the corresponding LR constant.
373  lr_log_message("Warning: Possible attempt to close a transaction that has not been opened!");
374  }
375  lr_save_int(status, saveparam);

457 char *y_calculate_actual_transaction_prefix(const char *transaction_prefix)
458 {
459  const char seperator[] = "_";
460  const int seperator_len = sizeof seperator - 1; // strlen(seperator);
461  int group_len = 0;
462  int prefix_len = strlen(transaction_prefix);
463  char *buffer;
464  size_t buffer_size;
465 
466  // y_virtual_user_group is set only if y_setup() is called.
467  // See y_loadrunner_utils.c
468  y_setup();
469  if( _y_add_group_to_trans )
470  {
471  group_len = strlen(y_virtual_user_group);
472  }
473 
474  // add room for the seperators
475  if(prefix_len > 0)
476  {
477  prefix_len += seperator_len;
478  }
479  if(group_len > 0)
480  {
481  group_len += seperator_len;
482  }
483 
484  // allocate memory -- note this needs to be free()'ed afterwards!
485  buffer_size = group_len + prefix_len + 1;
486  buffer = y_mem_alloc(buffer_size);
487  buffer[0] = '\0';
488 
489  // start concatenating things together
490  {
491  int len = 0;
492  if(group_len > 0)
493  {
494  len = snprintf(buffer, buffer_size, "%s%s", y_virtual_user_group, seperator);
495  }
496  if(prefix_len > 0)
497  {
498  snprintf(buffer + len, buffer_size-len,"%s%s", transaction_prefix, seperator);
499  }
500  }
501  return buffer;
502 }

524

618 // Note: This completely ignores the 'transaction_name' argument
619 // to retain compatibility with lr_end_transaction().
620 int y_end_transaction(char *transaction_name, int status)
621 {
622  char *trans_name = lr_eval_string("{y_current_transaction}");
623 
624  // Fire the transaction end trigger. For processing the results of
625  // complicated web_reg_find() / web_reg_save_param() statement
626  // collections that repeat for a group of transactions.
627  // This fires before the actual end of the transaction so that it can
628  // still influence the transaction end status.
629  int trigger_result = y_run_transaction_end_trigger();
630  if( status == LR_PASS && trigger_result != LR_PASS )
631  {
632  lr_error_message("Transaction end trigger did not return LR_PASS");
633  status = trigger_result;
634  }
635 
636  // Save the end status of this transaction. It won't be available after ending it.
637  y_save_transaction_end_status(trans_name, "y_last_transaction_status", status);
638 
639  // Debugging: report wasted time.
640  // People who implement their own triggers will have to do this themselves.

813 // Set up a series of chances that certain named steps will happen.
814 //
815 // The idea is that there is a .dat file listing names of steps and
816 // for each of those the chance that that step should be executed.
817 // The values in this .dat file are accessed 'on each occurrance' by the
818 // "step" and "stepchance" parameters.
819 //
820 // This function will iterate over the step parameter until it contains "END",
821 // and save each step chance to a seperate parameter with a name starting with
822 // 'step_chance_'.
823 //
824 // Before each step is executed a call is made to y_waterfall_random_weighted_continue()
825 // with the name of the step added. If the corresponding parameter was set up the
826 // value it contained is used as the weighted chance that the next part of the script
827 // will be executed.
828 //
829 // @author: Floris Kraak
830 //
831 void y_setup_step_waterfall()
832 {
833  char *step = lr_eval_string("{step}");
834  char *stepchance = lr_eval_string("{stepchance}");
835  char *head = "step_chance_";
836  char *tmp;
837 
838  while ( step && (strcmp(step, "END") != 0) )
839  {
840  size_t size = strlen(head) + strlen(step) +1;
841  tmp = y_mem_alloc(size);
842 
843  snprintf(tmp, size, "%s%s", head, step);
844 
845  lr_save_string(stepchance, tmp);
846  free(tmp);
847 
848  // We're hiding a for() loop here ;-)
849  step = lr_eval_string("{step}");
850  stepchance = lr_eval_string("{stepchance}");
851  }
852 }
