92 int VTS_process_returncode(int returncode)
93 {
94  char* errortext;
95 
120  // Encode the above information in a simple lookup table.
121  switch(returncode)
122  {
123  case VTCERR_OK:
124  errortext = "INFO: VTS command succeeded.";
125  lr_message(errortext);
126  return returncode;
127  break; // <-- never reached, but kept for readability.
128 
129  case VTCERR_INVALID_CONNECTION_INFO:
130  errortext = "Invalid connection info.";
131  break;
132 
133  case VTCERR_FAILED_TO_RESOLVE_ADDR:
134  errortext = "Failed to resolve address.";
135  break;
136 
137  case VTCERR_FAILED_TO_CREATE_SOCKET:
138  errortext = "Failed to create socket.";
139  break;
140 
141  case VTCERR_FAILED_TO_CONNECT:
142  errortext = "Failed to connect.";
143  break;
144 
145  case VTCERR_INCOMPLETE_REQUEST:
146  errortext = "Incomplete request.";
147  break;
148 
149  case VTCERR_FAILED_TO_RECV_RESPONSE:
150  errortext = "Failed to receive response.";
151  break;
152 
153  case VTCERR_INCOMPLETE_RESPONSE:
154  errortext = "Incomplete response.";
155  break;
156 
157  case VTCERR_RESPONSE_ARGS_UNMATCH:
158  errortext = "Received error code VTCERR_RESPONSE_ARGS_UNMATCH.";
159  break;
160 
161  case VTCERR_OPERATION_ERROR_BASE:
162  errortext = "Received error code VTCERR_OPERATION_ERROR_BASE.";
163  break;
164 
165  case VTCERR_SERVER_IS_BUSY:
166  errortext = "Server is busy. Go away.";
167  break;
168 
169  case VTCERR_CLIENT_REQUEST_ERROR:
170  errortext = "Client request error.";
171  break;
172 
173  default:
174  errortext = "Unknown VTC error code.";
175  break;
176  }
177 
178  // Report the error to the user. (The "All OK" case jumped out of this function earlier ..)
179  VTS_report_error(errortext);
180  // At this point it might be an idea to call lr_abort() or lr_exit(LR_EXIT_VUSER, LR_FAIL)
181 
182  return returncode;
183 }

229 // Voeg een waarde toe aan de onderkant van de tabel.
230 // Als de unique vlag groter dan 0 is gebeurt dat alleen onder voorwaarde dat de waarde niet al
231 // bestaat in de tabel.

236 int VTS_pushlast_with_flag(char* columnname, char* value, int unique)
237 {
238  int errorcode = 0;
239  char* errortext = "Write to VTS: OK.";
240  unsigned short status;
241  PVCI ppp = VTS_connect();
242 
243  if( ppp == -1 )
244  {
245  // VTS_connect() should have set the error message already.
246  return -1;
247  }
248 
249  if( unique > 0 )
250  {
251  errorcode = vtc_send_if_unique(ppp, columnname, value, &status);
252  }
253  else
254  {
255  errorcode = vtc_send_message(ppp, columnname, value, &status);
256  }
257  vtc_free(value);
258  VTS_disconnect();
259 
260  //lr_log_message("result: %d .... send message status: %d", errorcode, status);
261  if(VTS_process_returncode(errorcode) == VTCERR_OK)
262  {
263  if(status == 1)
264  {
265  // Success!
266  lr_message("INFO: Value pushed onto column successfully.");
267  }
268  else
269  {
270  errorcode = -2;
271  errortext = "Can not write to VTS: value (most likely) already exists in VTS.";
272  VTS_report_error(errortext);
273  }
274  }
275 
276  return errorcode;
277 }

346 int VTS_readRandom(char* columnname, char* ParameterName)
347 {
348  PVCI ppp = VTS_connect();
349  int rc = 0;
350  unsigned short status;
351  int errorcode = 0;
352  char* errortext;
353  int tablesize;
354  int rand_row;
355  char* value = NULL;
356 
357  if( ppp == -1 )
358  {
359  // VTS_connect() should have set the error message already.
360  return -1;
361  }
362 
363  if((rc = vtc_column_size(ppp, columnname, &tablesize)) != 0)
364  {
365  errorcode = rc;
366  errortext = "Can not determine column size. Error code %d.";
367  }
368 
369  //lr_log_message("tablesize: %d\n", tablesize);
370  rand_row = y_rand() % tablesize + 1;
371  if((rc = vtc_query_column(ppp, columnname, rand_row, &value)) != 0)
372  {
373  errortext = "******************** VTS Error - Query Return Code = %d";
374  errorcode = rc;
375  }
376  else
377  {
378  errortext = "INFO: VTS random read succeeded.";
379  lr_save_string(value,ParameterName);
380  }
381  vtc_free(value);
382  VTS_disconnect();
383 
384  if(errorcode != 0)
385  {
386  lr_error_message(errortext, errorcode);
387  }
388  lr_param_sprintf("VTS_ERROR_MESSAGE", errortext, errorcode);
389 
390  return errorcode;
391 }
