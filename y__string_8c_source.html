377 {
378  char *buffer;

382  //lr_log_message("y_split_str: original=%s, search=%s", original, search);
383 
384  if( posPtr == NULL )
385  {
386  // Copy the original to the left hand output buffer.

391

399  strncpy(right, start, strlen(start)+1); // Let's not forget to copy the null byte, too.
400  }

424 void y_split( const char *originalParameter, const char *separator, const char *leftParameter, const char *rightParameter)
425 {
426  char *item = y_get_parameter(originalParameter);
427  int len = strlen(item);
428 
429  if( len < strlen(separator) )
430  {
431  // Since the separator doesn't even fit in the item, searching is pointless.
432  // Store the original in the left hand parameter, the right hand is empty.
433  lr_save_string(originalParameter, leftParameter);
434  lr_save_string("", rightParameter);
435  return;
436 
437  }
438  else
439  {
440 
441  // Left hand side
442  // If the separator isn't found the full original string gets stored here.
443  // Don't forget the 0 byte at the end though..
444  char *left = y_mem_alloc(len+1);
445 
446  // Right hand side
447  // If the separator gets found in position 1 the remainder of the
448  // original string gets stored here (and left gets a zero-length string).
449  char *right = y_mem_alloc(len-strlen(separator)+1);
450 
451  // Start off with zero-length strings. We can safely assume
452  // both variables contain garbage when freshly allocated.
453  left[0] = '\0';
454  right[0] = '\0';
455 
456  // This is where the magic happens.
457  y_split_str(item, separator, left, right);
458 
459  // Store the results in parameters.
460  lr_save_string(left, leftParameter);
461  free(left);
462  lr_save_string(right, rightParameter);
463  free(right);
464  }
465 }

543 void y_replace( const char *parameter, const char *search, const char *replace )
544 {
545  int slen, rlen, plen; // lengte van search, replace, en basis string
546  int i = 0; // aantal replacements
547  int limit = 1000; // replacement limiet als replace > search
548 
549  char *c; // punt waar change moet beginnen
550  char *cend; // einde van de change = c+slen
551  char *last; // de \0 slotbyte van input string
552  char *buffer; // buffer voor bewerkingen
553  char *string; // originele string waar we in zoeken
554 
555  if ((search == NULL) || (strlen(search) <1))
556  return;
557  if (!search || !replace) return; // ongeldige search of_replace
558  if (!strcmp(search, replace)) return; // search == replace: geen wijziging
559 
560  slen = strlen (search); // de lengte van search
561  rlen = strlen (replace); // de lengte van replace
562 
563  string = y_get_parameter(parameter); // <-- memory allocated by loadrunner, too small if replace > search
564  plen = strlen(string);
565 
566  //lr_log_message("y_replace(%s, %s, %s) - slen %d, rlen %d, plen %d", parameter, search, replace, slen, rlen, plen);
567 
568  if ( rlen > slen)
569  {
570  // Reserve memory for -limit- replacements.
571  size_t size = plen + ((rlen-slen) * limit);
572  buffer = y_mem_alloc( size );
573  snprintf( buffer, size, "%s", string );
574  }
575  else
576  {
577  // No need to reserve memory
578  buffer = string;
579  }
580 
581  last = buffer + strlen(buffer) +1; // het einde van de string, de null byte
582  c = buffer; // initialiseer c - buffer moet intact blijven
583 
584  while (c = (char*) strstr(c, search)) // doorzoek de search string vanaf waar je gebleven bent
585  {
586  //lr_output_message("c:%s", c);
587 
588  i++;
589  if (slen != rlen) // zijn search en replace van verschillende lengte?
590  {
591  if( i >= limit )
592  {
593  lr_log_message("Unable to handle more than %d search-and-replaces, apologies for the inconvenience.", limit);
594  break;
595  }
596  cend = c+slen; // cend is de plaats van waaraf de string moet opschuiven
597  //lr_output_message("memmove(dest=%d, src=%d, size=%d)", (long)(c+rlen), (long)(cend), (long)(last-cend) );
598  memmove (c+rlen , cend, last-cend); // verplaats de rest van de string - als de geheugenberekening niet klopt gaat dit stuk!
599  //lr_output_message("memmove completed, result = %s", c);
600  last = last - slen + rlen; // en bereken het nieuwe eindpunt
601  }
602  memmove(c, replace, rlen); // execute the replacement
603  c += rlen; // start the next search at the point where the replacement ended
604  }
605  lr_save_string(buffer, parameter);
606 
607  if( rlen > slen )
608  {
609  free(buffer);
610  }
611 }

629 void y_remove_string_from_parameter(const char* paramName, const char* removeMe)
630 {
631  char *parameter;
632  char *removePtr;
633  int remlen;
634 
635  //lr_log_message("y_remove_string_from_parameter( remove:%s, parameter:%s )", removeMe, paramName);
636 
637  if(!removeMe || !*removeMe)
638  return;
639 
640  // fetch the contents of the parameter to change
641  parameter = y_get_parameter(paramName);
642  // removePtr is used to track our progress through this string
643  removePtr = parameter;
644  remlen = strlen(removeMe);
645 
646  // while we find occurrances of the string we're looking for
647  while ( removePtr = (char *)strstr( removePtr, removeMe ) )
648  {
649  // copy the characters between the end of the data we wish to remove and end-of-string
650  // to the place where we found our offending content.
651  char* origin = removePtr + remlen;
652  strncpy(removePtr, origin, (strlen(origin)+1));
653  }
654 
655  // store it in the original parameter
656  lr_save_string( parameter, paramName );
657 }
