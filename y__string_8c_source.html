418 void y_split( const char *originalParameter, const char *separator, const char *leftParameter, const char *rightParameter)
419 {
420  char *item = y_get_parameter(originalParameter);
421  int len = strlen(item);
422 
423  if( len < strlen(separator) )
424  {
425  // Since the separator doesn't even fit in the item, searching is pointless.
426  // Store the original in the left hand parameter, the right hand is empty.
427  lr_save_string(originalParameter, leftParameter);
428  lr_save_string("", rightParameter);
429  return;
430 
431  }
432  else
433  {
434 
435  // Left hand side
436  // If the separator isn't found the full original string gets stored here.
437  // Don't forget the 0 byte at the end though..
438  char *left = y_mem_alloc(len+1);
439 
440  // Right hand side
441  // If the separator gets found in position 1 the remainder of the
442  // original string gets stored here (and left gets a zero-length string).
443  char *right = y_mem_alloc(len-strlen(separator)+1);
444 
445  // Start off with zero-length strings. We can safely assume
446  // both variables contain garbage when freshly allocated.
447  left[0] = '\0';
448  right[0] = '\0';
449 
450  // This is where the magic happens.
451  y_split_str(item, separator, left, right);
452 
453  // Store the results in parameters.
454  lr_save_string(left, leftParameter);
455  free(left);
456  lr_save_string(right, rightParameter);
457  free(right);
458  }

537 void y_replace( const char *parameter, const char *search, const char *replace )
538 {
539  int slen, rlen, plen; // lengte van search, replace, en basis string
540  int i = 0; // aantal replacements
541  int limit = 1000; // replacement limiet als replace > search
542 
543  char *c; // punt waar change moet beginnen
544  char *cend; // einde van de change = c+slen
545  char *last; // de \0 slotbyte van input string
546  char *buffer; // buffer voor bewerkingen
547  char *string; // originele string waar we in zoeken
548 
549  if ((search == NULL) || (strlen(search) <1))
550  return;
551  if (!search || !replace) return; // ongeldige search of_replace
552  if (!strcmp(search, replace)) return; // search == replace: geen wijziging
553 
554  slen = strlen (search); // de lengte van search
555  rlen = strlen (replace); // de lengte van replace
556 
557  string = y_get_parameter(parameter); // <-- memory allocated by loadrunner, too small if replace > search
558  plen = strlen(string);
559 
560  //lr_log_message("y_replace(%s, %s, %s) - slen %d, rlen %d, plen %d", parameter, search, replace, slen, rlen, plen);
561 
562  if ( rlen > slen)
563  {
564  // Reserve memory for -limit- replacements.
565  size_t size = plen + ((rlen-slen) * limit);
566  buffer = y_mem_alloc( size );
567  snprintf( buffer, size, "%s", string );
568  }
569  else
570  {
571  // No need to reserve memory
572  buffer = string;
573  }
574 
575  last = buffer + strlen(buffer) +1; // het einde van de string, de null byte
576  c = buffer; // initialiseer c - buffer moet intact blijven
577 
578  while (c = (char*) strstr(c, search)) // doorzoek de search string vanaf waar je gebleven bent
579  {
580  //lr_output_message("c:%s", c);
581 
582  i++;
583  if (slen != rlen) // zijn search en replace van verschillende lengte?
584  {
585  if( i >= limit )
586  {
587  lr_log_message("Unable to handle more than %d search-and-replaces, apologies for the inconvenience.", limit);
588  break;
589  }
590  cend = c+slen; // cend is de plaats van waaraf de string moet opschuiven
591  //lr_output_message("memmove(dest=%d, src=%d, size=%d)", (long)(c+rlen), (long)(cend), (long)(last-cend) );
592  memmove (c+rlen , cend, last-cend); // verplaats de rest van de string - als de geheugenberekening niet klopt gaat dit stuk!
593  //lr_output_message("memmove completed, result = %s", c);
594  last = last - slen + rlen; // en bereken het nieuwe eindpunt
595  }
596  memmove(c, replace, rlen); // execute the replacement
597  c += rlen; // start the next search at the point where the replacement ended
598  }
599  lr_save_string(buffer, parameter);
600 
601  if( rlen > slen )
602  {
603  free(buffer);
604  }
605 }

731 void y_random_string_buffer_core(const char *parameter, int minimumLength, int maximumLength,
732  int minWordLength, int maxWordLength, char *characterSet)
733 {
734  char *buffer;
735  int charSetSize; // length of the characterSet
736  int length = 0;
737  int max = -1;
738 
739  char randomNumber;
740  int lettersInWord;
741 
742  charSetSize=strlen(characterSet);
743 
744  //lr_message("minimumLength %d -- maximumLength %d -- minWordLength %d -- maxWordLength %d",
745  // minimumLength, maximumLength, minWordLength, maxWordLength);
746 
747  // error checks - lots of code that saves us headaches later
748  if( minimumLength < 0 ) {
749  lr_error_message( "minimumLength less than 0 (%d)", minimumLength );
750  }
751  else if( maximumLength < 1 ) {
752  lr_error_message( "maximumLength less than 1 (%d)", maximumLength );
753  }
754  else if( maximumLength > (1024 * 1024) ) {
755  lr_error_message( "maximumLength too big (%d)", maximumLength );
756  }
757  else if( maximumLength < minimumLength ) {
758  lr_error_message( "minimumLength (%d) bigger than maximumLength (%d)", minimumLength, maximumLength );
759  }
760  else if(maximumLength > minimumLength) {
761  // Not an error
762  max = y_rand_between(minimumLength, maximumLength);
763  lr_log_message("Max: %d", max);
764  }
765  else if(maximumLength == minimumLength) {
766  // Not an error either
767  max = maximumLength;
768  }
769  else {
770  lr_error_message("This can never happen. If we reach this point it's a bug.");
771  }
772 
773  // if we got an error
774  if( max < 0 )
775  {
776  lr_set_transaction_status(LR_FAIL);
777  lr_exit(LR_EXIT_ITERATION_AND_CONTINUE, LR_FAIL);
778  }
779 
780  // get memory for the buffer
781  buffer = (char *)y_mem_alloc( max +1 );
782  // note: if this fails y_mem_alloc() aborts the script, so no error handling needed.
783 
784  while( length < max )
785  {
786 // lr_message("Length: %d max: %d", length, max);
787 // lettersInWord = ((y_rand() % 8) + 2);
788  if( maxWordLength == 0 )
789  {
790  lettersInWord = maximumLength;
791  }
792  else
793  {
794  lettersInWord = y_rand_between(minWordLength, maxWordLength);
795  if( lettersInWord < 0 )
796  {
797  lr_error_message( "y_rand_between() returned an errorcode (%d)", lettersInWord );
798  lr_set_transaction_status(LR_FAIL);
799  lr_exit(LR_EXIT_ITERATION_AND_CONTINUE, LR_FAIL);
800  }
801  }
802 
803  while( lettersInWord-- && (length < (max)) )
804  {
805  randomNumber = (char) (y_rand() % charSetSize);
806  buffer[length++] = characterSet[randomNumber];
807  }
808 
809  if( maxWordLength != 0 )
810  {
811  if( length < max -1 )
812  {
813  buffer[length++] = ' ';
814  }
815  }
816  }
817  buffer[max] = '\0';
818 
819  lr_save_string(buffer, parameter);
820  free(buffer);
821 }
