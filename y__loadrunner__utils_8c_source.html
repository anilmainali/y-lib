390  if( (fwrite(szBuf, nLength, 1, fp)) < 0 )

393  result = -2;

399  }
400

405  if( result >= 0 ) // if no error occured during writing, fclose() failed, return -3.
406  result = -3;
407  }
408 
409  // Free the parameter content buffer.
410  lr_eval_string_ext_free(&szBuf);
411  return result;

800 double y_think_time_for_rampup_ext(const int rampup_period, double TPS_initial, double TPS_max, const int virtual_users)
801 {
802  static double test_start_time = 0; // Test starttime in seconds since 1 jan 1970.
803  static double previous_time = 0; // Timestamp of the last call to this, after think time.
804  double time_passed; // Elapsed time since test start, in seconds.
805  double response_time; // Elapsed time since previous call.
806  double current_time = y_get_current_time(); // Current time, in seconds since 1 jan 1970.
807 
808  // Initialisation.
809  // On the first call we store the current time as the test start time and the end time of the previous call.
810  if( test_start_time < 1 )
811  {
812  test_start_time = current_time;
813  previous_time = current_time;
814  }
815 
816  // Calculate how much time has passed since test start and the previous call.
817  time_passed = current_time - test_start_time;
818  response_time = current_time - previous_time;
819 
820  // Debugging//
821  lr_log_message("TT calculation: starttime %f, current time %f, previous time %f, virtual_users %d, rampup_period %d",
822  test_start_time, current_time, previous_time, virtual_users, rampup_period);
823 
824  {
825  double TT, TPS_target;
826  double factor = time_passed / rampup_period; // multiplication factor for the target load, based on rampup.
827 
828  if( factor > 1 ) // after rampup load should be stable.
829  factor = 1;
830  else if( factor < 0.03 ) // To prevent extreme thinktimes the first 3% of the rampup thinktime is calculated as if 3% of the time has passed.
831  factor = 0.03;
832 
833  // double TPS_max = virtual_users * (1/response_time); // this is the absolute theoretical maximum, which we aren't using right now.
834  TPS_target = ((TPS_max - TPS_initial) * factor) + TPS_initial; // Apply the ramp up to get the target TPS.
835  TT = (virtual_users / TPS_target) - response_time; // Apply the base formula to get the resulting think time
836 
837  // Debugging.
838  lr_log_message("TT: %f, time_passed: %f, factor %f, response_time %f, TPS_init %f, TPS_max %f, TPS_target %f",
839  TT, time_passed, factor, response_time, TPS_initial, TPS_max, TPS_target);
840 
841  lr_user_data_point("y_thinktime", TT);
842  if( TT > 0 )
843  lr_think_time(TT);
844 
845  // Store the old time in our backup location.
846  //previous_time = current_time;
847  // Do a new time measurement..
848  //ftime(&ts);
849  //current_time = ts.time + (ts.millitm / 1000);
850 
851  // We measure the response time by comparing previous_time to a new measurement later, but that measurement
852  // will have to be schewed a tad to account for rounding in the windows sleep() call used by lr_think_time().
853  //{
854  //double time_slept = current_time - previous_time;
855  //double deviation = TT - time_slept;
856  //previous_time = current_time + deviation;
857  //lr_log_message("Slept %f sec, deviation %f, current_time %f with deviation added: %f", time_slept, deviation, current_time, previous_time);
858  //}
859 
860  // All of the above can be rewritten to:
861  previous_time = current_time + TT; // Note that the "current time" actually is the timestamp from *before* we slept.
862 
863  return TT;
864  }
865 }

896 y_execute_shell_command(char* command, int debug)
897 {
898  const int buffer_size = 10240; // 10 KB;
899  char buffer[10240]; // allocate memory for the output of the command.
900  // Has to be hardcoded because this compiler is stupid about const keywords and I do not want to use #define because of potential side effects. -- FBK
901  long fp; // file/stream pointer
902  int count; // number of characters that have been read from the stream.
903  char *token;
904  char *command_evaluated = lr_eval_string(command);
905 
906  lr_save_string("-- command not yet executed --", "command_result");
907  lr_log_message("Executing command: %s", command_evaluated);
908 
909  fp = popen(command_evaluated, "r");
910  if (fp == NULL) {
911  lr_error_message("Error opening stream.");
912  return -1;
913  }
914 
915  buffer[0] = '\0'; // Clear the buffer before we try to fill it - Prevents the previous command output from showing up in here. -- FBK
916  count = fread(buffer, sizeof(char), buffer_size, fp); // read up to 10KB
917  if (feof(fp) == 0)
918  {
919  lr_error_message("Did not reach the end of the input stream when reading. Try increasing buffer_size.");
920  pclose(fp);
921  return -1;
922  }
923  if (ferror(fp))
924  {
925  lr_error_message ("I/O error during read.");
926  pclose(fp);
927  return -1;
928  }
929  count = fread(buffer, sizeof(char), (sizeof buffer) - 1, fp);
930  lr_save_var(buffer, count, 0, "command_output");
931 
932  // Split the stream at each newline character, and save them to a parameter array.
933  token = (char*) strtok(buffer, "\n"); // Get the first token
934 
935  if (token == NULL) {
936  lr_save_string("", "command_result");
937  }
938  else
939  {
940  lr_save_string(token, "command_result"); // First token saved
941 
942  if(debug)
943  {
944  char param_buf[10]; // buffer to hold the parameter name.
945  int i = 1;
946 
947  while (token != NULL) { // While valid tokens are returned
948  sprintf(param_buf, "output_%d", i);
949  lr_save_string(token, param_buf);
950  i++;
951  token = (char*) strtok(NULL, "\n");
952  }
953  lr_save_int(i-1, "output_count");
954 
955  // Print all values of the parameter array.
956  for (i=1; i<=lr_paramarr_len("output"); i++) {
957  lr_output_message("Parameter value: %s", lr_paramarr_idx("output", i));
958  }
959  }
960  }
961 
962  pclose(fp);
963  return 0;
964 }

997 int y_errorcheck(int ok)
998 {
999  static int errorcount = 0; // Static means this value will not be reset to zero when a new iteration starts.
1000 
1001  // Configuration from virtual user group command line or runtime attribute settings.
1002  long enabled_attrib = lr_get_attrib_long("errorcheck_enabled");
1003  long error_limit_attrib = lr_get_attrib_long("errorcheck_limit");
1004  long pause_time_attrib = lr_get_attrib_long("errorcheck_pause_time_seconds");
1005 
1006  // Defaults
1007  int enabled = enabled_attrib >= 0 ? enabled_attrib : 0; // default off (Should not be used for breaktesting, should be ON for peakload or endurance testing.)
1008  int error_limit = error_limit_attrib >= 0 ? error_limit_attrib : 10; // number of failed iterations before firing
1009  int pause_time = pause_time_attrib >= 0 ? pause_time_attrib : 900; // how much time to wait (in seconds) when this fires.
1010 
1011  //lr_log_message("y_errorcheck() enabled: %d (%d), error_limit: %d (%d), pause_time: %d (%d)", enabled, enabled_attrib, error_limit, error_limit_attrib, pause_time, pause_time_attrib );
1012 
1013  if( !enabled )
1014  return 0;
1015  if (ok)
1016  {
1017  errorcount = 0;
1018  }
1019  else
1020  {
1021  if (errorcount >= error_limit)
1022  {
1023  lr_error_message("y_errorcheck(): Too many errors occurred. Pausing %d seconds.", pause_time);
1024  lr_set_transaction(lr_eval_string("---TOO MANY ERRORS - THROTTLING LOAD---"), 0, LR_FAIL);
1025  lr_force_think_time(pause_time); // alternative would be calling lr_abort();
1026  }
1027  if (errorcount)
1028  lr_log_message("Number of failed iterations: %d", errorcount);
1029 
1030  lr_user_data_point( "y_errorcheck_errorcount", errorcount++);
1031  }
1032 
1033  return 0; // Adding this function to the run logic has the same effect as y_errorcheck(0);
1034 }
