90 int y_rand_in_sliding_window(int lowerbound, int upperbound, int rand_max)
91 {

93

95  {
96  lr_error_message("y_rand_in_sliding_window called with nonsensical arguments: ( 0 <= %d < %d <= %d ) == FALSE",
97  lowerbound, upperbound, rand_max);
98  return -1;
99  }
100 
101  roll = y_rand_between(0, rand_max);

104  return 1;
105  }
106 
107  return 0;

128 int y_rand_between(int lowerbound, int upperbound)
129 {
130  int roll;
131 
132  if( (lowerbound < 0) || (lowerbound > upperbound) || ((upperbound - lowerbound) == 0) )
133  {
134  lr_error_message("y_rand() called with negative or nonsensical arguments. Lowerbound should be less than upperbound!");
135  return -1; // Note to self: This is a classic case for standard error codes.
136  }
137  roll = y_rand() % ((upperbound + 1 - lowerbound)) + lowerbound;
138  return roll;
139 }

393 int y_workdays_from_today(int workdays)
394 {
395  int weekday, weeksOffset, weekstart;
396  int i = 0;
397  int result = workdays;
398  //int debugOffset = 0;
399 
400  // debugging loop
401  //for(debugOffset = 0; debugOffset < 13; debugOffset++) {
402  // result = debugOffset;
403 
404  //lr_log_message("--- result start %d ---", result);
405 
406  // Determine what day of the week today falls into.
407  lr_save_datetime("%w", DATE_NOW, "weekdayToday");
408  weekstart = atoi(lr_eval_string("{weekdayToday}"));
409  //lr_log_message("--- weekstart = %d ---", weekstart);
410 
411  weeksOffset = result / 5;
412  //lr_log_message("weeksOffset %d", weeksOffset);
413  result += (2 * weeksOffset);
414  //lr_log_message("Adding extra weeks weekends adds up to %d", result);
415 
416  // Determine what day of the week our target day falls into.
417  lr_save_datetime("%w", DATE_NOW + result*(ONE_DAY), "weekdayFuture");
418  weekday = atoi(lr_eval_string("{weekdayFuture}"));
419  //lr_log_message("--- weekday = %d ---", weekday);
420 
421  // Look at each day between the day of the week that our count started on,
422  // and the day of our target date. Shift the target date backwards if we find
423  // a weekend.
424  i = weekstart;
425  do {
426  // Weekend rollover
427  if( i > 6) {
428  i = 0;
429  }
430  // Weekend day found
431  if( i == 0 || i == 6 ) {
432  //lr_log_message("i = %d, adding 1 day", i);
433  result++;
434  }
435  i++;
436  } while( i != (weekday+1) );
437 
438  // Add another day if our target day falls on a saturday.
439  // The search loop accounted for the saturday itself, but saturdays tend to be
440  // followed by sundays ..
441  if ( weekday == 6 ) {
442  //lr_log_message("weekday = %d, adding 1 day", weekday);
443  result++;
444  }
445 
446  //lr_log_message( "############ Final day offset = %d ############", result);
447 
448  //} // end debugging loop
449  return result;
450 }

734 double y_think_time_for_rampup_ext(const int rampup_period, double TPS_initial, double TPS_max, const int virtual_users)
735 {
736  static double test_start_time = 0; // Test starttime in seconds since 1 jan 1970.
737  static double previous_time = 0; // Timestamp of the last call to this, after think time.
738  double time_passed; // Elapsed time since test start, in seconds.
739  double response_time; // Elapsed time since previous call.
740  double current_time = y_get_current_time(); // Current time, in seconds since 1 jan 1970.
741 
742  // Initialisation.
743  // On the first call we store the current time as the test start time and the end time of the previous call.
744  if( test_start_time < 1 )
745  {
746  test_start_time = current_time;
747  previous_time = current_time;
748  }
749 
750  // Calculate how much time has passed since test start and the previous call.
751  time_passed = current_time - test_start_time;
752  response_time = current_time - previous_time;
753 
754  // Debugging//
755  lr_log_message("TT calculation: starttime %f, current time %f, previous time %f, virtual_users %d, rampup_period %d",
756  test_start_time, current_time, previous_time, virtual_users, rampup_period);
757 
758  {
759  double TT, TPS_target;
760  double factor = time_passed / rampup_period; // multiplication factor for the target load, based on rampup.
761 
762  if( factor > 1 ) // after rampup load should be stable.
763  factor = 1;
764  else if( factor < 0.03 ) // To prevent extreme thinktimes the first 3% of the rampup thinktime is calculated as if 3% of the time has passed.
765  factor = 0.03;
766 
767  // double TPS_max = virtual_users * (1/response_time); // this is the absolute theoretical maximum, which we aren't using right now.
768  TPS_target = ((TPS_max - TPS_initial) * factor) + TPS_initial; // Apply the ramp up to get the target TPS.
769  TT = (virtual_users / TPS_target) - response_time; // Apply the base formula to get the resulting think time
770 
771  // Debugging.
772  lr_log_message("TT: %f, time_passed: %f, factor %f, response_time %f, TPS_init %f, TPS_max %f, TPS_target %f",
773  TT, time_passed, factor, response_time, TPS_initial, TPS_max, TPS_target);
774 
775  lr_user_data_point("y_thinktime", TT);
776  if( TT > 0 )
777  lr_think_time(TT);
778 
779  // Store the old time in our backup location.
780  //previous_time = current_time;
781  // Do a new time measurement..
782  //ftime(&ts);
783  //current_time = ts.time + (ts.millitm / 1000);
784 
785  // We measure the response time by comparing previous_time to a new measurement later, but that measurement
786  // will have to be schewed a tad to account for rounding in the windows sleep() call used by lr_think_time().
787  //{
788  //double time_slept = current_time - previous_time;
789  //double deviation = TT - time_slept;
790  //previous_time = current_time + deviation;
791  //lr_log_message("Slept %f sec, deviation %f, current_time %f with deviation added: %f", time_slept, deviation, current_time, previous_time);
792  //}
793 
794  // All of the above can be rewritten to:
795  previous_time = current_time + TT; // Note that the "current time" actually is the timestamp from *before* we slept.
796 
797  return TT;
798  }
799 }
