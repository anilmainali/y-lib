110 long y_rand()
111 {
112  // Have we initialized the random seed yet?
113  static int _y_random_seed_initialized = 0;
114 
115  if(!_y_random_seed_initialized)
116  {
117  int seed, tm, rnd;
118  y_setup();
119 
120  // Seed the random number generator for later use.
121  // To make it random enough for our purposes mix in the vuser id and the adress of the vuser group name.
122  // In case the script itself already initialized the random number generator, use a random number from
123  // there as well.
124 
125  tm = (int)time(NULL);
126  rnd = rand();
127 
128  //lr_log_message("Seed values - time: %d, y_virtual_user_id: %d, y_virtual_user_group: %d, rand: %d", tm, y_virtual_user_id, (int)y_virtual_user_group, rnd);
129  seed = tm%10000 + y_virtual_user_id + ((int)y_virtual_user_group)%1000 + rnd%1000;
130  //lr_log_message("Initialising random seed: %d", seed);
131  srand( seed );
132  _y_random_seed_initialized = 1;
133  }
134 
135  {
136  // Because rand() does not return numbers above 32767 and we want to get at least 30 of the 31 bits
137  // of randomness that a long affords us we are going to roll multiple numbers and basically
138  // concatenate them together using bit shifts.
139  //
140  // ( If we were to go to 32 bits this function would return negative numbers, which would be undesirable
141  // because it will break people's expectations of what rand() does.)
142 
143  long result = rand() << 15 | rand();
144  //lr_log_message("y_rand: 30 random bits = %d, RAND_MAX = %d", result, RAND_MAX);
145 
146  // Doing a third call to rand() just to get 1 bit of entropy isn't really effici√´nt ..
147  //result = (result << 1) | (rand() & 0x0000000000000001); // add another bit and we're done.
148  lr_log_message("y_rand: final random roll = %x, RAND_MAX = %d", result, RAND_MAX);
149 
150  return result;
151  }
152 }

164 static unsigned long y_hash_sdbm(char* str)
165 {
166  unsigned long hash = 0;
167  int c;
168  while (c = *str++)
169  hash = c + (hash << 6) + (hash << 16) - hash;
170  return hash;
171 }

252 void y_random_string_buffer_core(const char *parameter, int minimumLength, int maximumLength,
253  int minWordLength, int maxWordLength, char *characterSet)
254 {
255  char *buffer;
256  int charSetSize; // length of the characterSet
257  int length = 0;
258  int max = -1;
259 
260  char randomNumber;
261  int lettersInWord;
262 
263  charSetSize=strlen(characterSet);
264 
265  //lr_message("minimumLength %d -- maximumLength %d -- minWordLength %d -- maxWordLength %d",
266  // minimumLength, maximumLength, minWordLength, maxWordLength);
267 
268  // error checks - lots of code that saves us headaches later
269  if( minimumLength < 0 ) {
270  lr_error_message( "minimumLength less than 0 (%d)", minimumLength );
271  }
272  else if( maximumLength < 1 ) {
273  lr_error_message( "maximumLength less than 1 (%d)", maximumLength );
274  }
275  else if( maximumLength > (1024 * 1024) ) {
276  lr_error_message( "maximumLength too big (%d)", maximumLength );
277  }
278  else if( maximumLength < minimumLength ) {
279  lr_error_message( "minimumLength (%d) bigger than maximumLength (%d)", minimumLength, maximumLength );
280  }
281  else if(maximumLength > minimumLength) {
282  // Not an error
283  max = y_rand_between(minimumLength, maximumLength);
284  lr_log_message("Max: %d", max);
285  }
286  else if(maximumLength == minimumLength) {
287  // Not an error either
288  max = maximumLength;
289  }
290  else {
291  lr_error_message("This can never happen. If we reach this point it's a bug.");
292  }
293 
294  // if we got an error
295  if( max < 0 )
296  {
297  lr_set_transaction_status(LR_FAIL);
298  lr_exit(LR_EXIT_ITERATION_AND_CONTINUE, LR_FAIL);
299  }
300 
301  // get memory for the buffer
302  buffer = (char *)y_mem_alloc( max +1 );
303  // note: if this fails y_mem_alloc() aborts the script, so no error handling needed.
304 
305  while( length < max )
306  {
307 // lr_message("Length: %d max: %d", length, max);
308 // lettersInWord = ((y_rand() % 8) + 2);
309  if( maxWordLength == 0 )
310  {
311  lettersInWord = maximumLength;
312  }
313  else
314  {
315  lettersInWord = y_rand_between(minWordLength, maxWordLength);
316  if( lettersInWord < 0 )
317  {
318  lr_error_message( "y_rand_between() returned an errorcode (%d)", lettersInWord );
319  lr_set_transaction_status(LR_FAIL);
320  lr_exit(LR_EXIT_ITERATION_AND_CONTINUE, LR_FAIL);
321  }
322  }
323 
324  while( lettersInWord-- && (length < (max)) )
325  {
326  randomNumber = (char) (y_rand() % charSetSize);
327  buffer[length++] = characterSet[randomNumber];
328  }
329 
330  if( maxWordLength != 0 )
331  {
332  if( length < max -1 )
333  {
334  buffer[length++] = ' ';
335  }
336  }
337  }
338  buffer[max] = '\0';
339 
340  lr_save_string(buffer, parameter);
341  free(buffer);
342 }

453 int y_rand_in_sliding_window(int lowerbound, int upperbound, int randMax)
454 {
455  int roll;
456 
457  if( (0>lowerbound) || (lowerbound>upperbound) || (upperbound > randMax) || (randMax <= 0))
458  {
459  lr_error_message("y_rand_in_sliding_window called with nonsensical arguments: ( 0 <= %d < %d <= %d ) == FALSE",
460  lowerbound, upperbound, randMax);
461  return -1;
462  }
463 
464  roll = y_rand_between(0, randMax);
465  if( (roll >= lowerbound) && (roll <= upperbound) )
466  {
467  return 1;
468  }
469 
470  return 0;
471 }

488 int y_rand_between(int lowerbound, int upperbound)
489 {
490  int roll;
491 
492  if( (lowerbound < 0) || (lowerbound > upperbound) || ((upperbound - lowerbound) == 0) )
493  {
494  lr_error_message("y_rand() called with negative or nonsensical arguments. Lowerbound should be less than upperbound!");
495  return -1; // Note to self: This is a classic case for standard error codes.
496  }
497  roll = y_rand() % ((upperbound + 1 - lowerbound)) + lowerbound;
498  return roll;
499 }

803  }

973 // For simulating situations with limited amounts of connections on the client side.
974 //
975 // In such a case we cannot use regular vuser based rampups, so instead we have to gradually lower the thinktime to get a similar effect.
976 // This implements a "think time" function that will use response time calculations and a TPS target to get simulate such a ramp up.
977 //
978 // Base formula:
979 //
980 // TT = threads/TPS(target) - response time,
981 //
982 // while increasing TPS linearly until rampup_period has passed, until either think time reaches zero or TPS_max is reached.
983 // Note that "response time" is actually "time passed since the previous call to this function". This includes wasted time, for example.
984 //
985 // For best effect, call this once in vuser_init and once for each transaction, preferably by overloading lr_think_time()
986 // with your own version that calls this (or the simplified wrapper, below) every time, like so:
987 //
988 // void my_little_think_time()
989 // {
990 // y_think_time_for_rampup(1800, 10);
991 // }
992 //
993 //
994 // Parameters ( = recommended default ):
995 // const int rampup_period = 1800; - How long the rampup should take, in seconds.
996 // double TPS_initial = 0.1; - Start rampup with this many transactions / sec for this transaction, across all virtual users.
997 // Do not set this too low or thinktime values may get really large.
998 // double TPS_max = 10; - End rampup with this many transactions / sec in total, with 'virtual_users' users. This cannot be higher than 1/average response time.
999 // Note that the actual TPS achieved is highly dependant on what mix of thinktimes and transactions are used.
1000 // Notably, this assumes you only have one transaction after each call to this function, and that each call uses the same parameters.
1001 // const int virtual_users = 1; - How many virtual users the script is using. If you use "1", you can just use TPS / virtual user for the initial target TPS.
1002 //
1003 double y_think_time_for_rampup_ext(const int rampup_period, double TPS_initial, double TPS_max, const int virtual_users)
1004 {
1005  static double test_start_time = 0; // Test starttime in seconds since 1 jan 1970.
1006  static double previous_time = 0; // Timestamp of the last call to this, after think time.
1007  double time_passed; // Elapsed time since test start, in seconds.
1008  double response_time; // Elapsed time since previous call.
1009  double current_time = y_get_current_time(); // Current time, in seconds since 1 jan 1970.
1010 
1011  // Initialisation.
1012  // On the first call we store the current time as the test start time and the end time of the previous call.
1013  if( test_start_time < 1 )
1014  {
1015  test_start_time = current_time;
1016  previous_time = current_time;
1017  }
1018 
1019 
1020  // Calculate how much time has passed since test start and the previous call.
1021  time_passed = current_time - test_start_time;
1022  response_time = current_time - previous_time;
1023 
1024  // Debugging//
1025  lr_log_message("TT calculation: starttime %f, current time %f, previous time %f, virtual_users %d, rampup_period %d",
1026  test_start_time, current_time, previous_time, virtual_users, rampup_period);
1027 
1028 
1029  {
1030  double TT;
1031  double TPS_target;
1032  double factor = time_passed / rampup_period; // multiplication factor for the target load, based on rampup.
1033 
1034  if( factor > 1 ) // after rampup load should be stable.
1035  factor = 1;
1036  else if( factor < 0.03 ) // To prevent extreme thinktimes the first 3% of the rampup thinktime is calculated as if 3% of the time has passed.
1037  factor = 0.03;
1038 
1039  // double TPS_max = virtual_users * (1/response_time); // this is the absolute theoretical maximum, which we aren't using right now.
1040  TPS_target = ((TPS_max - TPS_initial) * factor) + TPS_initial; // Apply the ramp up to get the target TPS.
1041  TT = (virtual_users / TPS_target) - response_time; // Apply the base formula to get the resulting think time
1042 
1043  // Debugging.
1044  lr_log_message("TT: %f, time_passed: %f, factor %f, response_time %f, TPS_init %f, TPS_max %f, TPS_target %f",
1045  TT, time_passed, factor, response_time, TPS_initial, TPS_max, TPS_target);
1046 
1047  lr_user_data_point("ThinkTime", TT);
1048  if( TT > 0 )
1049  lr_think_time(TT);
1050 
1051  // Store the old time in our backup location.
1052  //previous_time = current_time;
1053  // Do a new time measurement..
1054  //ftime(&ts);
1055  //current_time = ts.time + (ts.millitm / 1000);
1056 
1057  // We measure the response time by comparing previous_time to a new measurement later, but that measurement
1058  // will have to be schewed a tad to account for rounding in the windows sleep() call used by lr_think_time().
1059  //{
1060  //double time_slept = current_time - previous_time;
1061  //double deviation = TT - time_slept;
1062  //previous_time = current_time + deviation;
1063  //lr_log_message("Slept %f sec, deviation %f, current_time %f with deviation added: %f", time_slept, deviation, current_time, previous_time);
1064  //}
1065 
1066  // All of the above can be rewritten to:
1067  previous_time = current_time + TT; // Note that the "current time" actually is the timestamp from *before* we slept.
1068 
1069  return TT;
1070  }
1071 }
