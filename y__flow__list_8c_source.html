36 // Wrap the function pointer in a structure that
37 // gives the flow a name and a weight - how much chance it

40 // The chance of a specific flow getting executed is 1 in

58 // Given a flow list of a specified length, calculate what number all flow
59 // weights added together add up to.
60 // Do not call directly unless you have checked the flow list for sanity beforehand.
61 int y_calc_flow_weight_total(y_flow flow_list[], int flow_count)
62 {
63  int i, total = 0;
64  for(i=0; i < flow_count; i++)
65  {
66  y_flow* flow = &flow_list[i];
67  total += flow->weight;
68  }
69  lr_log_message("y_flow: Combined total of weights is: %d", total);
70  return total;
71 }

74 //
75 // Choose a flow from a list of flows.
76 // Returns a pointer to a randomly chosen flow, or NULL if
77 // somehow the weights don't add up to 100 (or more).
78 //
79 // The flow_count argument should exactly match the number of flows
80 // in the array or the script might blow up with MEMORY_ACCESS_VIOLATION errors.
81 //
82 // Do not call directly unless you have checked the flow list for sanity beforehand.
83 y_flow* y_choose_flow(y_flow flow_list[], int flow_count)
84 {
85  //y_flow **flow_list = flow_list_ptr;
86  int i, lowerbound = 0;
87  int cursor = 0;
88  int roll = y_rand() % y_calc_flow_weight_total(flow_list, flow_count);
89 
90  lr_log_message("Roll: %d", roll);
91 
92  for(i=0; i < flow_count; i++)
93  {
94  y_flow *flow = &flow_list[i];
95  int weight = flow->weight;
96  cursor += weight;
97 
98  lr_log_message("weight cursor: %d", cursor);
99  if(roll < cursor)
100  {
101  return flow;
102  }
103  }
104  return NULL;
105 }

107 //
108 // Execute a chosen flow, provided the pointer to the flow passed in is not NULL.
109 // This will setup a transaction to measure the duration of the transaction as well.
110 //
111 void y_exec_flow(y_flow *chosen_flow)
112 {
113  if(chosen_flow == NULL)
114  {
115  lr_log_message("Warning: Cannot execute NULL flow.");
116  }
117  else if(chosen_flow->name == NULL)
118  {
119  lr_log_message("Warning: Cannot execute flow without a name!");
120  }
121  else if(chosen_flow->function == NULL)
122  {
123  lr_log_message("Warning: Cannot execute NULL flow function for flow %s",
124  chosen_flow->name);
125  }
126  else
127  {
128  y_flow_func *flow_function = chosen_flow->function;
129 
130  // Run the flow.
131  flow_function();
132  }
133 }
