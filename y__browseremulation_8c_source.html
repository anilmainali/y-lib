179  browser->name = browser_name;

183  // but really we're just filling out the remaining fields of the browser struct.
184  {
185  int error = 0;
186  char* browser_chance = y_get_parameter_or_null(browser_chance_param);
187  char* browser_max_connections_per_host = y_get_parameter_or_null(browser_max_connections_per_host_param);

241  lr_abort();

347 y_browser* y_choose_browser_from_list(y_browser* browser_list_head )
348 {
349  int i, lowerbound, cursor = 0;
350  int max = y_browser_list_chance_total;

352  long roll;
353 
354  if( max < 1 )
355  {
356  lr_error_message("y_browseremulation.c: Browser list not initialised before call to y_choose_browser_from_list(). Cannot choose, ignoring.");
357  return NULL;
358  }

360  // The upper bound of the rand() function is determined by the RAND_MAX constant.
361  // RAND_MAX is hardcoded in loadrunner to a value of exactly 32767.
362  // In fact, it doesn't even get #defined by loadrunner as as the C standard mandates.
363  // (y_core.c does that instead, now ..)
364  //
365  // y_rand() depends on rand() for it's output, so that cannot go above 32767.
366  // Update: y_rand() uses Y_RAND_MAX now - slightly over 1 billion.
367  // Unfortunately, the list with weights that we use has numbers bigger than that.
368  //
369  // So we'll need to get a bit creative, and multiply the result of y_rand() with
370  // the maximum (the total of the browser chances) divided by Y_RAND_MAX to scale things up again.
371  // (At the cost of precision ..)
372  //
373  // Update: y_rand() now returns a 31 bit number, giving it an upper bound of 2147483647 (Y_RAND_MAX).
374  // That should reduce the need for this code by a bit ..
375  //
376 
377  roll = y_rand() % max;
378  //lr_log_message("max = %d, RAND_MAX = %d, roll %d", max, RAND_MAX, roll);
379  if( Y_RAND_MAX < max)
380  {
381  roll = roll * ((max / RAND_MAX));
382  }
383  //lr_log_message("Roll: %d", roll);
384 
385  for( browser = browser_list_head; browser != NULL; browser = browser->next)
386  {
387  cursor += browser->chance;
388  // lr_log_message("Chance cursor: %d for browser: %s", cursor, browser->name);
389 
390  if(roll <= cursor)
391  {
392  //lr_log_message("Chosen browser:");
393  //y_log_browser(browser);
394  return browser;
395  }
396  }
397  // This code should be unreachable.
398  lr_error_message("y_browseremulation.c: Roll result out of bounds: roll: %d, cursor: %d, max %d", roll, cursor, max);
399  return browser;
400 }

409 void y_emulate_browser(const y_browser* browser)
410 {
411  char str_max_connections[12];
412  char str_max_connections_per_host[12];
413  int max_connections;
414 
415  if( browser == NULL )
416  {
417  lr_error_message("y_browser_emulation.c: Attempt to emulate the NULL browser: Ignored.");
418  return;
419  }
420 
421  // Debugging purposes .
422  lr_log_message("Emulating browser:");
423  y_log_browser(browser);
424 
425  // Loadrunner doesn't accept values higher than 50 for this sockets option, so we'll just log it and set it to 50.
426  max_connections = browser->max_connections;
427  if( max_connections > 50 )
428  {
429  lr_log_message("y_browser_emulation.c: Loadrunner does not support using more than 50 browser connections. Using 50 connections instead of %d.", max_connections);
430  max_connections = 50;
431  }
432 
433  snprintf(str_max_connections, sizeof str_max_connections, "%d", max_connections);
434  snprintf(str_max_connections_per_host, sizeof str_max_connections_per_host, "%d", browser->max_connections_per_host);
435 
436  // Now finally set the correct settings for the chosen browser:
437  web_set_sockets_option("MAX_CONNECTIONS_PER_HOST", str_max_connections_per_host);
438  web_set_sockets_option("MAX_TOTAL_CONNECTIONS", str_max_connections);
439  web_add_auto_header("User-Agent", browser->user_agent_string);
440 }
